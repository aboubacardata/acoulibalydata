# Server logic for updating map based on inputs
server <- function(input, output, session) {

  # Reactive data filter
  filtered_data <- reactive({
    city_data %>%
      filter(
        date >= input$date_range[1],
        date <= input$date_range[2],
        performance_score < input$threshold
      )
  })

  # Render Leaflet Map
  output$map <- renderLeaflet({
    leaflet(filtered_data()) %>%
      addTiles() %>%
      addCircleMarkers(
        ~lng, ~lat,
        radius = ~sqrt(incidents),
        color = "#FF4B4B",
        popup = ~paste("Zone:", zone_id, "
Score:", performance_score)
      )
  })
  
  # CRM Alert Trigger
  observeEvent(input$trigger_alert, {
    send_crm_alert(input$selected_zone)
    showNotification("Alert sent to CRM workflow.")
  })
}
